void SendDHTData(void)
{
	DHT_SensorData* pDHT = DHT_ReadData(MT3620_GPIO0);

	char tempBuffer[20];
	
	int len = snprintf(tempBuffer, 20, "%0.2f", pDHT->TemperatureCelsius);
	if (len > 0) {
		SendTelemetry("Temperature", tempBuffer);
	}
}
